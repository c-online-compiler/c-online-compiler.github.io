calculatrice.c 
#include <stdio.h>

int main() {
    double a, b;
    char op;

    printf("Entrez une expression : ");
    scanf("%lf %c %lf", &a, &op, &b);

    switch(op) {
        case '+':
            printf("%lf\n", a + b);
            break;
        case '-':
            printf("%lf\n", a - b);
            break;
        case '*':
            printf("%lf\n", a * b);
            break;
        case '/':
            if (b != 0)
                printf("%lf\n", a / b);
            else
                printf("Erreur : division par zéro\n");
            break;
        default:
            printf("Erreur : opération inconnue\n");
    }

    return 0;
}
