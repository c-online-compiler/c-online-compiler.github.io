calculatrice.cpp 
#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "Entrez une expression : ";
    cin >> a >> op >> b;

    switch(op) {
        case '+':
            cout << a + b << endl;
            break;
        case '-':
            cout << a - b << endl;
            break;
        case '*':
            cout << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << a / b << endl;
            else
                cout << "Erreur : division par zéro" << endl;
            break;
        default:
            cout << "Erreur : opération inconnue" << endl;
    }

    return 0;
}
