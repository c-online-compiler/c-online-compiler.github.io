calculator.cpp 
#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "수식 입력: ";
    cin >> a >> op >> b;

    switch(op) {
        case '+':
            cout << a + b << endl;
            break;
        case '-':
            cout << a - b << endl;
            break;
        case '*':
            cout << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << a / b << endl;
            else
                cout << "오류: 0으로 나눌 수 없습니다" << endl;
            break;
        default:
            cout << "오류: 알 수 없는 연산" << endl;
    }

    return 0;
}
