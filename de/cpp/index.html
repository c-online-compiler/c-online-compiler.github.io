Grundlegendes C++-Programm zur Demonstration der Stream-Ausgabe.

rechner.cpp 
#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "Ausdruck eingeben: ";
    cin >> a >> op >> b;

    switch(op) {
        case '+':
            cout << a + b << endl;
            break;
        case '-':
            cout << a - b << endl;
            break;
        case '*':
            cout << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << a / b << endl;
            else
                cout << "Fehler: Division durch Null" << endl;
            break;
        default:
            cout << "Fehler: Unbekannte Operation" << endl;
    }

    return 0;
}
