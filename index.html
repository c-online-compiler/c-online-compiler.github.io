calculator.c 
#include <stdio.h>

int main() {
    double a, b;
    char op;

    printf("Введите выражение: ");
    scanf("%lf %c %lf", &a, &op, &b);

    switch(op) {
        case '+':
            printf("%lf\n", a + b);
            break;
        case '-':
            printf("%lf\n", a - b);
            break;
        case '*':
            printf("%lf\n", a * b);
            break;
        case '/':
            if (b != 0)
                printf("%lf\n", a / b);
            else
                printf("Ошибка: деление на ноль\n");
            break;
        default:
            printf("Ошибка: неизвестная операция\n");
    }

    return 0;
}
