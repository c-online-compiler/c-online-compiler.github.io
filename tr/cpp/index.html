calculator.cpp 
#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "İşlemi girin: ";
    cin >> a >> op >> b;

    switch(op) {
        case '+':
            cout << a + b << endl;
            break;
        case '-':
            cout << a - b << endl;
            break;
        case '*':
            cout << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << a / b << endl;
            else
                cout << "Hata: sıfıra bölme" << endl;
            break;
        default:
            cout << "Hata: bilinmeyen işlem" << endl;
    }

    return 0;
}
