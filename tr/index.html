hesap_makinesi.c 
#include <stdio.h>

int main() {
    double a, b;
    char op;

    printf("Bir ifade girin: ");
    scanf("%lf %c %lf", &a, &op, &b);

    switch(op) {
        case '+':
            printf("%lf\n", a + b);
            break;
        case '-':
            printf("%lf\n", a - b);
            break;
        case '*':
            printf("%lf\n", a * b);
            break;
        case '/':
            if (b != 0)
                printf("%lf\n", a / b);
            else
                printf("Hata: sıfıra bölme\n");
            break;
        default:
            printf("Hata: bilinmeyen işlem\n");
    }

    return 0;
}
