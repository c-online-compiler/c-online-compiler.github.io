calculator.cpp 
#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "Enter expression: ";
    cin >> a >> op >> b;

    switch(op) {
        case '+':
            cout << a + b << endl;
            break;
        case '-':
            cout << a - b << endl;
            break;
        case '*':
            cout << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << a / b << endl;
            else
                cout << "Error: division by zero" << endl;
            break;
        default:
            cout << "Error: unknown operation" << endl;
    }

    return 0;
}
