calculator.c 
#include <stdio.h>

int main() {
    double a, b;
    char op;

    printf("Enter an expression: ");
    scanf("%lf %c %lf", &a, &op, &b);

    switch(op) {
        case '+':
            printf("%lf\n", a + b);
            break;
        case '-':
            printf("%lf\n", a - b);
            break;
        case '*':
            printf("%lf\n", a * b);
            break;
        case '/':
            if (b != 0)
                printf("%lf\n", a / b);
            else
                printf("Error: division by zero\n");
            break;
        default:
            printf("Error: unknown operation\n");
    }

    return 0;
}
